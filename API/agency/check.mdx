代理商查询用户流量

    **post** &nbsp; &nbsp;&nbsp; &nbsp;&nbsp; &nbsp;/account/queryTraffic

    **Body**
    ---
    <ParamField path="appKey" type="string" required>
        appKey密钥
    </ParamField>
    <ParamField path="params" type="object" required>
        动态参数集合
        <Card>
            <ParamField path="&nbsp;username" type="string" required>
                &nbsp;&nbsp;用户名
            </ParamField>
        </Card>
        
    </ParamField>

    **Response**
    ---
    <ResponseField name="code" type="enum<integer>" defalut="default: 0">
        状态码，0为操作成功

        Available options: 0, -1
    </ResponseField>
    <ResponseField name="data" type="object">
            <Card>
            <ResponseField name="&nbsp;fullTraffic" type="integer">
                &nbsp;&nbsp;总流量
            </ResponseField>
            <ResponseField name="&nbsp;traffic" type="integer">
                &nbsp;&nbsp;剩余流量
            </ResponseField>
        </Card>
    </ResponseField>
    <ResponseField name="msg" type="string">
        响应消息
    </ResponseField>

        ```bash Go
            package main

            import (
                "fmt"
                "strings"
                "net/http"
                "io/ioutil"
            )

            func main() {

                url := "https://apiv1.cakeip.com/openapi/account/queryTraffic"

                payload := strings.NewReader("{\n  \"appKey\": \"<string>\",\n  \"params\": {\n    \"username\": \"<string>\"\n  }\n}")

                req, _ := http.NewRequest("POST", url, payload)

                req.Header.Add("Content-Type", "application/json")

                res, _ := http.DefaultClient.Do(req)

                defer res.Body.Close()
                body, _ := ioutil.ReadAll(res.Body)

                fmt.Println(res)
                fmt.Println(string(body))

            }
        ```
